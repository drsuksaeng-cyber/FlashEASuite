#include <Trade/Trade.mqh>

class CStrategyManager
  {
private:
   CStrategyBase* m_strategies[];
   int               m_total_strategies;
   CTrade            m_trade;
   
   // ‡πÄ‡∏Å‡∏ì‡∏ë‡πå‡∏Å‡∏≤‡∏£‡πÇ‡∏´‡∏ß‡∏ï‡∏ú‡πà‡∏≤‡∏ô (‡πÄ‡∏ä‡πà‡∏ô ‡∏Ñ‡∏∞‡πÅ‡∏ô‡∏ô‡∏£‡∏ß‡∏°‡∏ï‡πâ‡∏≠‡∏á‡πÄ‡∏Å‡∏¥‡∏ô 50)
   double            m_vote_threshold;

public:
   CStrategyManager()
     {
      m_vote_threshold = 50.0;
      m_trade.SetExpertMagicNumber(999000);
     }
     
   ~CStrategyManager()
     {
      for(int i=0; i<m_total_strategies; i++) delete m_strategies[i];
      ArrayResize(m_strategies, 0);
     }

   void Initialize()
     {
      // ‡πÄ‡∏û‡∏¥‡πà‡∏° Strategy ‡πÄ‡∏Ç‡πâ‡∏≤‡∏™‡∏†‡∏≤
      AddStrategy(new CStrategySpike()); 
      // AddStrategy(new CStrategyGrid()); // ‡∏≠‡∏ô‡∏≤‡∏Ñ‡∏ï
     }
     
   void AddStrategy(CStrategyBase* strat)
     {
      m_total_strategies = ArraySize(m_strategies);
      ArrayResize(m_strategies, m_total_strategies + 1);
      m_strategies[m_total_strategies] = strat;
     }

   // --- ‡∏£‡∏±‡∏ö‡∏ô‡πâ‡∏≥‡∏´‡∏ô‡∏±‡∏Å‡∏à‡∏≤‡∏Å Python (Online Mode) ---
   void UpdateWeights(double w_spike, double w_grid)
     {
      if(m_total_strategies > 0) m_strategies[0].SetWeight(w_spike); // Spike
      // if(m_total_strategies > 1) m_strategies[1].SetWeight(w_grid);  // Grid
     }

   // --- ‡∏ó‡∏≥‡∏á‡∏≤‡∏ô‡∏ó‡∏∏‡∏Å Tick ---
   void OnTickLogic()
     {
      double total_buy_score = 0;
      double total_sell_score = 0;
      
      // 1. ‡∏ß‡∏ô‡∏•‡∏π‡∏õ‡∏ñ‡∏≤‡∏°‡∏Ñ‡∏∞‡πÅ‡∏ô‡∏ô‡∏à‡∏≤‡∏Å‡∏ó‡∏∏‡∏Å Strategy
      for(int i=0; i<ArraySize(m_strategies); i++)
        {
         double raw_score = m_strategies[i].GetScore(); // -100 ‡∏ñ‡∏∂‡∏á 100
         double weight = m_strategies[i].GetWeight();   // 0.0 ‡∏ñ‡∏∂‡∏á 1.0
         
         // ‡∏Ñ‡∏∞‡πÅ‡∏ô‡∏ô‡∏ñ‡πà‡∏ß‡∏á‡∏ô‡πâ‡∏≥‡∏´‡∏ô‡∏±‡∏Å
         double weighted_score = raw_score * weight;
         
         if(weighted_score > 0) total_buy_score += weighted_score;
         if(weighted_score < 0) total_sell_score += MathAbs(weighted_score);
        }
        
      // 2. ‡∏ï‡∏±‡∏î‡∏™‡∏¥‡∏ô‡πÉ‡∏à (Vote Resolution)
      // (‡∏ï‡∏±‡∏ß‡∏≠‡∏¢‡πà‡∏≤‡∏á Logic: ‡∏ù‡∏±‡πà‡∏á‡πÑ‡∏´‡∏ô‡∏ä‡∏ô‡∏∞‡∏Ç‡∏≤‡∏î ‡πÅ‡∏•‡∏∞‡∏ú‡πà‡∏≤‡∏ô‡πÄ‡∏Å‡∏ì‡∏ë‡πå)
      
      if(total_buy_score > m_vote_threshold && total_buy_score > total_sell_score)
        {
         ExecuteTrade(ORDER_TYPE_BUY);
        }
      else if(total_sell_score > m_vote_threshold && total_sell_score > total_buy_score)
        {
         ExecuteTrade(ORDER_TYPE_SELL);
        }
     }
     
   void ExecuteTrade(ENUM_ORDER_TYPE type)
     {
      // ‡∏™‡πà‡∏á‡∏Ñ‡∏≥‡∏™‡∏±‡πà‡∏á‡∏ã‡∏∑‡πâ‡∏≠‡∏Ç‡∏≤‡∏¢‡∏£‡∏ß‡∏°...
      m_trade.PositionOpen(_Symbol, type, 0.01, 0, 0, 0, "Council_Vote");
     }
  };
